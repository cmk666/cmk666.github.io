statement

给定 nnn 个不为原点且不重复的点 (xi,yi)(x_i,y_i)(xi​,yi​)，问原点可以看到多少个不被别的点遮挡的点。

n≤2×105n\le2\times10^5n≤2×105，∣xi∣,∣yi∣≤104|x_i|,|y_i|\le10^4∣xi​∣,∣yi​∣≤104。

solution

等价于求不同的 xy\dfrac xyyx​ 个数，除以 gcd⁡\gcdgcd 丢进 map 里就行。

code 

#include <bits/stdc++.h>

int main() {
	std::cin.tie(nullptr)->sync_with_stdio(false);

C

statement

从 nnn 个长为 mmm 的且只含大写字母的字符串中选 kkk 个重排成一个字符串，使得其字典序最小。

1≤k≤n≤3001\le k\le n\le3001≤k≤n≤300，1≤m≤3001\le m\le3001≤m≤300。

solution

按 (cntA,cntB,⋯ ,cntZ)(cnt_A,cnt_B,\cdots,cnt_Z)(cntA​,cntB​,⋯,cntZ​) 降序排序，取前 kkk 个，把对应字母的出现次数加起来，直接构造即可。

code 

#include <bits/stdc++.h>

int main() {
	std::cin.tie(nullptr)->sync_with_stdio(false);
	int n, m, k;
	std::cin >> n >> m >> k;
	std::vector<std::array<int, 26>> cnt(n, std::array<int, 26>());
	for (int i = 0; i < n; i++) {
		std::string s;
		std::cin >> s;
		for (char j : s) cnt[i][j - 'A']++;
	}
	std::vector<int> id(n);
	std::iota(id.begin(), id.end(), 0);
	std::sort(id.begin(), id.end(), [&](int x, int y) {
		for (int i = 0; i < 26; i++) {
			if (cnt[x][i] > cnt[y][i]) return true;
			if (cnt[x][i] < cnt[y][i]) return false;
		}
		return false;
	});
	std::array<int, 26> res = {};
	for (int i = 0; i < k; i++) {
		for (int j = 0; j < 26; j++) res[j] += cnt[id[i]][j];
	}
	for (int i = 0; i < 26; i++) {
		for (int j = 0; j < res[i]; j++) std::cout << (char)(i + 'A');
	}
	std::cout << '\n';
}

D

statement

有 nnn 个物品自上而下挂成一列，第 iii 个物品有分数 aia_iai​。

两个人轮流打，打一个物品会获得这个物品的分数，且另一个人会获得下面挂着的物品的分数总和，然后这些物品全部消失；或选择不打，另一个人获得剩下所有物品分数总和，游戏结束。

求先手最优策略下的最大分数。

1≤n≤2×1061\le n\le2\times10^61≤n≤2×106，∣ai∣≤109|a_i|\le10^9∣ai​∣≤109。

solution

考虑 dp，设 fif_ifi​ 表示只剩最顶上 iii 个物品时的分差，则 fi=max⁡j<i{−fj+aj+1−∑k=j+2iak}f_i=\max_{j<i}\{-f_j+a_{j+1}-\sum_{k=j+2}^ia_k\}fi​=maxj<i​{−fj​+aj+1​−∑k=j+2i​ak​}。

记前缀和 si=∑j=1iajs_i=\sum_{j=1}^ia_jsi​=∑j=1i​aj​，则 fi=max⁡j<i{−fj+aj+1−(si−sj+1)}f_i=\max_{j<i}\{-f_j+a_{j+1}-(s_i-s_{j+1})\}fi​=maxj<i​{−fj​+aj+1​−(si​−sj+1​)}。转移的时候开个变量，维护前缀的 max⁡{−fj+aj+1+sj+1}\max\{-f_j+a_{j+1}+s_{j+1}\}max{−fj​+aj+1​+sj+1​} 就行了。

总复杂度 O(n)O(n)O(n)。

code 

#include <bits/stdc++.h>

using i64 = long long;

int main() {
	std::cin.tie(nullptr)->sync_with_stdio(false);
	int n;
	std::cin >> n;
	std::vector<i64> a(n);
	for (i64 &i : a) std::cin >> i;
	std::vector<i64> s(n);
	std::partial_sum(a.begin(), a.end(), s.begin());
	std::vector<i64> dp(n);
	i64 maxn = std::max(0ll, s[0] + a[0]);
	for (int i = 0; i < n; i++) {
		dp[i] = maxn - s[i];
		if (i < n) maxn = std::max(maxn, s[i + 1] + a[i + 1] - dp[i]);
	}
	i64 sum = s.back(), dif = dp.back();
	std::cout << (sum + dif) / 2 << '\n';
}

E

statement

把 nnn 个数 a1,a2,⋯ ,ana_1,a_2,\cdots,a_na1​,a2​,⋯,an​ 分成若干个连续段，使得相邻两组长度不同，且每组的异或和的和最大。

1≤n≤1061\le n\le10^61≤n≤106，0≤ai≤1090\le a_i\le10^90≤ai​≤109。

solution

显然的，异或肯定比加法不优，因此长度尽量小。

发现长度 >4>4>4 可以拆开且不会矛盾，直接 dp，fi,jf_{i,j}fi,j​ 表示到第 iii 个，这组长度为 jjj 的最大和。转移直接枚举前面那组的长度即可。

带个 42=164^2=1642=16 倍常数的 O(n)O(n)O(n)。

code 

#include <bits/stdc++.h>

using i64 = long long;

int main() {
	std::cin.tie(nullptr)->sync_with_stdio(false);
	int n;
	std::cin >> n;
	std::vector<int> a(n);
	for (int &i : a) std::cin >> i;
	std::vector<int> s(n);
	std::partial_sum(a.begin(), a.end(), s.begin(), std::bit_xor());
	std::vector<std::vector<i64>> dp(n, std::vector<i64>(5, -1e18));
	for (int i = 0; i < n; i++) {
		for (int j = 1; j <= 4; j++) {
			if (j == i + 1) {
				dp[i][j] = s[i];
			} else if (j > i + 1) {
				dp[i][j] = -1e18;
			} else {
				i64 maxn = -1e18;
				for (int k = 1; k <= 4; k++) {
					if (j != k) maxn = std::max(maxn, dp[i - j][k]);
				}
				dp[i][j] = maxn + (s[i] ^ s[i - j]);
			}
		}
	}
	std::cout << *std::max_element(dp.back().begin(), dp.back().end()) << '\n';
}

F

statement

给一个长为 nnn 的排列 aaa 和一个位置 kkk。

两个人轮流操作，一次操作可以交换 ai,aja_i,a_jai​,aj​，当且仅当 1≤i<j≤k∧ai<aj1\le i<j\le k\land a_i<a_j1≤i<j≤k∧ai​<aj​，或者 k≤i<j≤n∧ai>ajk\le i<j\le n\land a_i>a_jk≤i<j≤n∧ai​>aj​，无法操作者输。

问最优策略下谁会赢。

1≤n≤2×1051\le n\le2\times10^51≤n≤2×105。

solution

显然的，一次交换必然改变整个序列逆序对奇偶性。以下记 fx=a1,a2,⋯ ,axf_x=a_1,a_2,\cdots,a_xfx​=a1​,a2​,⋯,ax​ 的顺序对个数，gx=ax,ax+1,⋯ ,ang_x=a_x,a_{x+1},\cdots,a_ngx​=ax​,ax+1​,⋯,an​ 的逆序对个数。

若 ak=1a_k=1ak​=1，两边独立，先手赢当且仅当 2∤(fk−1+gk+1)2\nmid(f_{k-1}+g_{k+1})2∤(fk−1​+gk+1​)。

下面假设 111 在 <k<k<k 处，111 在 >k>k>k 处是同理的。

若 2∤(fk+gk+1)2\nmid(f_k+g_{k+1})2∤(fk​+gk+1​)，则先手可以把 111 和 aka_kak​ 交换，使得两边独立的同时 2∣(fk−1+gk+1)2\mid(f_{k-1}+g_{k+1})2∣(fk−1​+gk+1​)，后手必败。

若 2∣(fk+gk+1)2\mid(f_k+g_{k+1})2∣(fk​+gk+1​)，则两个人都不能改变 fk+gk+1f_k+g_{k+1}fk​+gk+1​ 的奇偶性，不然另一个人交换 111 和 kkk 这个人就寄了。因此一定是换 aka_kak​ 和某个在 aka_kak​ 右边且奇偶性与 aka_kak​ 不同的数。

记 m=min⁡i>k{ai}m=\min_{i>k}\{a_i\}m=mini>k​{ai​}，那么除非 m>akm>a_km>ak​，此时先手换不了一点，后手获胜；否则最终肯定会把 aka_kak​ 换成 mmm，所以先手赢当且仅当换了奇数次，aka_kak​ 的奇偶性最终发生了变化，即一开始时 aka_kak​ 与 mmm 的奇偶性不同。

O(nlog⁡n)O(n\log n)O(nlogn)，瓶颈在于逆序对。

code 

#include <bits/stdc++.h>

using i64 = long long;

int main() {
	std::cin.tie(nullptr)->sync_with_stdio(false);
	int n, k;
	std::cin >> n >> k;
	k--;
	std::vector<int> a(n);
	for (int &i : a) std::cin >> i;
	int one = std::find(a.begin(), a.end(), 1) - a.begin();
	const auto inversion = [&](int l, int r, bool left) {
		std::vector<i64> fenwick(n + 1, 0);
		i64 res = 0;
		if (left) {
			for (int i = l; i <= r; i++) {
				for (int pos = a[i]; pos; pos ^= pos & -pos) res += fenwick[pos];
				for (int pos = a[i]; pos <= n; pos += pos & -pos) fenwick[pos]++;
			};
		} else {
			for (int i = r; i >= l; i--) {
				for (int pos = a[i]; pos; pos ^= pos & -pos) res += fenwick[pos];
				for (int pos = a[i]; pos <= n; pos += pos & -pos) fenwick[pos]++;
			};
		}
		return res;
	};
	if (one == k) {
		i64 total = inversion(0, k - 1, 1) + inversion(k + 1, n - 1, 0);
		std::cout << (total & 1 ? "Minchan\n" : "Junee\n");
		return 0;
	} else if (one < k) {
		i64 total = inversion(0, k, 1) + inversion(k + 1, n - 1, 0);
		if (total & 1) {
			std::cout << "Minchan\n";
			return 0;
		}
		int mi = *min_element(a.begin() + k + 1, a.end());
		if (mi > a[k]) {
			std::cout << "Junee\n";
			return 0;
		}
		std::cout << ((mi ^ a[k]) & 1 ? "Minchan\n" : "Junee\n");
		return 0;
	} else {
		i64 total = inversion(0, k - 1, 1) + inversion(k, n - 1, 0);
		if (total & 1) {
			std::cout << "Minchan\n";
			return 0;
		}
		int mi = *min_element(a.begin(), a.begin() + k + 1);
		if (mi > a[k]) {
			std::cout << "Junee\n";
			return 0;
		}
		std::cout << ((mi ^ a[k]) & 1 ? "Minchan\n" : "Junee\n");
		return 0;
	}
}

G

statement

给定 nnn 个物品，每个物品有三个价值 wi,vi,piw_i,v_i,p_iwi​,vi​,pi​，你要选其中 kkk 个，使得 ∑w+max⁡v+min⁡p\sum w+\max v+\min p∑w+maxv+minp 最小，并构造方案。

1≤k≤n≤2×1051\le k\le n\le2\times10^51≤k≤n≤2×105，0≤wi,vi,pi≤1090\le w_i,v_i,p_i\le10^90≤wi​,vi​,pi​≤109。

solution

按照 vvv 从小到大排序，枚举 vvv，那么问题变成：每次加一个物品 w,pw,pw,p，要求钦定这个物品必选的同时 ∑w+min⁡p\sum w+\min p∑w+minp 最小。

维护一个大小为 kkk 的集合 SSS，里面存的是当前的最优选择。初始时里面装满 (∞,∞)(\infin,\infin)(∞,∞)。

考虑加入一个物品 (w,p)(w,p)(w,p)，由于钦定其必选，很自然的想到贪心，把 SSS 中 www 最大的物品扔到换成这个。但是这样也有可能存在删去这个后 min⁡p\min pminp 反而变大使得答案不优的情况，此时删去 SSS 中 www 第二大的物品换掉即可，这样 min⁡p\min pminp 不会变大。当然也可以直接枚举这两种选择。

如果插入一个物品后答案比原来更劣，则可以在更新答案后直接把操作撤销回去。

输出方案是容易的。

O(nlog⁡n)O(n\log n)O(nlogn)，实现细节很多。

code 

#include <bits/stdc++.h>
 
using i64 = long long;
 
int main() {
	std::cin.tie(nullptr)->sync_with_stdio(false);
	int n, k;
	std::cin >> n >> k;
	std::vector<std::tuple<int, int, int, int>> a(n);
	for (int i = 0; i < n; i++) {
		auto &[w, v, p, id] = a[i];
		std::cin >> w >> v >> p;
		id = i;
	}
	if (k == 1) {
		std::pair<i64, int> ans = std::make_pair((i64)1e18, -1);
		for (auto [w, v, p, id] : a) {
			ans = std::min(ans, std::make_pair(0ll + w + v + p, id));
		}
		std::cout << ans.first << '\n' << ans.second + 1 << '\n';
		return 0;
	}
	auto b = a;
	std::sort(a.begin(), a.end(), [](auto x, auto y) {
		return std::get<1>(x) < std::get<1>(y);
	});
	std::set<std::pair<int, int>, std::greater<std::pair<int, int>>> ws;
	std::set<std::pair<int, int>> ps;
	for (int i = 0; i < k; i++) {
		b.emplace_back((int)2e9, (int)2e9, (int)2e9, n + i);
		ws.emplace((int)2e9, n + i);
		ps.emplace((int)2e9, n + i);
	}
	i64 cur = (i64)2e9 * (k + 1);
	auto ans = std::make_tuple((i64)1e18, -1, -1);
	for (auto [w, v, p, id] : a) {
		const auto del = [&](int idx) {
			i64 sum = cur - std::get<0>(b[idx]) - ps.begin()->first;
			ws.erase(std::make_pair(std::get<0>(b[idx]), idx));
			ps.erase(std::make_pair(std::get<2>(b[idx]), idx));
			ws.emplace(w, id);
			ps.emplace(p, id);
			sum += w + ps.begin()->first;
			int ret = ps.begin()->second;
			ws.erase(std::make_pair(w, id));
			ps.erase(std::make_pair(p, id));
			ws.emplace(std::get<0>(b[idx]), idx);
			ps.emplace(std::get<2>(b[idx]), idx);
			return std::make_tuple(sum, idx, ret);
		};
		int del1 = ws.begin()->second, del2 = next(ws.begin())->second;
		auto nw = std::min(del(del1), del(del2));
		ans = std::min(ans, std::make_tuple(get<0>(nw) + v, id, get<2>(nw)));
		if (get<0>(nw) < cur) {
			cur = get<0>(nw);
			ws.erase(std::make_pair(std::get<0>(b[get<1>(nw)]), get<1>(nw)));
			ps.erase(std::make_pair(std::get<2>(b[get<1>(nw)]), get<1>(nw)));
			ws.emplace(w, id);
			ps.emplace(p, id);
		}
	}
	auto [sum, mxv, mnp] = ans;
	std::cout << sum << '\n';
	std::vector<std::pair<int, int>> mx;
	for (auto [w, v, p, id] : a) {
		if (id != mxv && id != mnp) {
			if (v <= std::get<1>(b[mxv]) && p >= std::get<2>(b[mnp])) {
				mx.emplace_back(w, id);
			}
		}
	}
	std::sort(mx.begin(), mx.end());
	std::vector<int> res;
	for (int i = 0; i < (mxv == mnp ? k - 1 : k - 2); i++) {
		res.push_back(mx[i].second);
	}
	res.push_back(mxv);
	if (mxv != mnp) res.push_back(mnp);
	for (int i : res) std::cout << i + 1 << " \n"[i == res.back()];
}
