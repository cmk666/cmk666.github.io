C

statement

从 nnn 个长为 mmm 的且只含大写字母的字符串中选 kkk 个重排成一个字符串，使得其字典序最小。

1≤k≤n≤3001\le k\le n\le3001≤k≤n≤300，1≤m≤3001\le m\le3001≤m≤300。

solution

按 (cntA,cntB,⋯ ,cntZ)(cnt_A,cnt_B,\cdots,cnt_Z)(cntA​,cntB​,⋯,cntZ​) 降序排序，取前 kkk 个，把对应字母的出现次数加起来，直接构造即可。

code 

#include <bits/stdc++.h>

int main() {
	std::cin.tie(nullptr)->sync_with_stdio(false);
	int n, m, k;
	std::cin >> n >> m >> k;
	std::vector<std::array<int, 26>> cnt(n, std::array<int, 26>());
	for (int i = 0; i < n; i++) {
		std::string s;
		std::cin >> s;
		for (char j : s) cnt[i][j - 'A']++;
	}
	std::vector<int> id(n);
	std::iota(id.begin(), id.end(), 0);
	std::sort(id.begin(), id.end(), [&](int x, int y) {
		for (int i = 0; i < 26; i++) {
			if (cnt[x][i] > cnt[y][i]) return true;
			if (cnt[x][i] < cnt[y][i]) return false;
		}
		return false;
	});
	std::array<int, 26> res = {};
	for (int i = 0; i < k; i++) {
		for (int j = 0; j < 26; j++) res[j] += cnt[id[i]][j];
	}
	for (int i = 0; i < 26; i++) {
		for (int j = 0; j < res[i]; j++) std::cout << (char)(i + 'A');
	}
	std::cout << '\n';
}

D

statement

有 nnn 个物品自上而下挂成一列，第 iii 个物品有分数 aia_iai​。

两个人轮流打，打一个物品会获得这个物品的分数，且另一个人会获得下面挂着的物品的分数总和，然后这些物品全部消失；或选择不打，另一个人获得剩下所有物品分数总和，游戏结束。

求先手最优策略下的最大分数。

1≤n≤2×1061\le n\le2\times10^61≤n≤2×106，∣ai∣≤109|a_i|\le10^9∣ai​∣≤109。

solution

考虑 dp，设 fif_ifi​ 表示只剩最顶上 iii 个物品时的分差，则 fi=max⁡j<i{−fj+aj+1−∑k=j+2iak}f_i=\max_{j<i}\{-f_j+a_{j+1}-\sum_{k=j+2}^ia_k\}fi​=maxj<i​{−fj​+aj+1​−∑k=j+2i​ak​}。

记前缀和 si=∑j=1iajs_i=\sum_{j=1}^ia_jsi​=∑j=1i​aj​，则 fi=max⁡j<i{−fj+aj+1−(si−sj+1)}f_i=\max_{j<i}\{-f_j+a_{j+1}-(s_i-s_{j+1})\}fi​=maxj<i​{−fj​+aj+1​−(si​−sj+1​)}。转移的时候开个变量，维护前缀的 max⁡{−fj+aj+1+sj+1}\max\{-f_j+a_{j+1}+s_{j+1}\}max{−fj​+aj+1​+sj+1​} 就行了。

总复杂度 O(n)O(n)O(n)。

code 

#include <bits/stdc++.h>

using i64 = long long;

int main() {
	std::cin.tie(nullptr)->sync_with_stdio(false);
	int n;
	std::cin >> n;
	std::vector<i64> a(n);
	for (i64 &i : a) std::cin >> i;
	std::vector<i64> s(n);
	std::partial_sum(a.begin(), a.end(), s.begin());
	std::vector<i64> dp(n);
	i64 maxn = std::max(0ll, s[0] + a[0]);
	for (int i = 0; i < n; i++) {
		dp[i] = maxn - s[i];
		if (i < n) maxn = std::max(maxn, s[i + 1] + a[i + 1] - dp[i]);
	}
	i64 sum = s.back(), dif = dp.back();
	std::cout << (sum + dif) / 2 << '\n';
}

E

statement

把 nnn 个数 a1,a2,⋯ ,ana_1,a_2,\cdots,a_na1​,a2​,⋯,an​ 分成若干个连续段，使得相邻两组长度不同，且每组的异或和的和最大。

1≤n≤1061\le n\le10^61≤n≤106，0≤ai≤1090\le a_i\le10^90≤ai​≤109。

solution

显然的，异或肯定比加法不优，因此长度尽量小。

发现长度 >4>4>4 可以拆开且不会矛盾，直接 dp，fi,jf_{i,j}fi,j​ 表示到第 iii 个，这组长度为 jjj 的最大和。转移直接枚举前面那组的长度即可。

带个 42=164^2=1642=16 倍常数的 O(n)O(n)O(n)。

code 

#include <bits/stdc++.h>

using i64 = long long;

int main() {
	std::cin.tie(nullptr)->sync_with_stdio(false);
	int n;
	std::cin >> n;
	std::vector<int> a(n);
	for (int &i : a) std::cin >> i;
	std::vector<int> s(n);
	std::partial_sum(a.begin(), a.end(), s.begin(), std::bit_xor());
	std::vector<std::vector<i64>> dp(n, std::vector<i64>(5, -1e18));
	for (int i = 0; i < n; i++) {
		for (int j = 1; j <= 4; j++) {
			if (j == i + 1) {
				dp[i][j] = s[i];
			} else if (j > i + 1) {
				dp[i][j] = -1e18;
			} else {
				i64 maxn = -1e18;
				for (int k = 1; k <= 4; k++) {
					if (j != k) maxn = std::max(maxn, dp[i - j][k]);
				}
				dp[i][j] = maxn + (s[i] ^ s[i - j]);
			}
		}
	}
	std::cout << *std::max_element(dp.back().begin(), dp.back().end()) << '\n';
}

F

statement

给一个长为 nnn 的排列 aaa 和一个位置 kkk。

两个人轮流操作，一次操作可以交换 ai,aja_i,a_jai​,aj​，当且仅当 1≤i<j≤k∧ai<aj1\le i<j\le k\land a_i<a_j1≤i<j≤k∧ai​<aj​，或者 k≤i<j≤n∧ai>ajk\le i<j\le n\land a_i>a_jk≤i<j≤n∧ai​>aj​，无法操作者输。

问最优策略下谁会赢。

1≤n≤2×1051\le n\le2\times10^51≤n≤2×105。

solution

显然的，一次交换必然改变整个序列逆序对奇偶性。以下记 fx=a1,a2,⋯ ,axf_x=a_1,a_2,\cdots,a_xfx​=a1​,a2​,⋯,ax​ 的顺序对个数，gx=ax,ax+1,⋯ ,ang_x=a_x,a_{x+1},\cdots,a_ngx​=ax​,ax+1​,⋯,an​ 的逆序对个数。

若 ak=1a_k=1ak​=1，两边独立，先手赢当且仅当 2∤(fk−1+gk+1)2\nmid(f_{k-1}+g_{k+1})2∤(fk−1​+gk+1​)。

下面假设 111 在 <k<k<k 处，111 在 >k>k>k 处是同理的。

若 2∤(fk+gk+1)2\nmid(f_k+g_{k+1})2∤(fk​+gk+1​)，则先手可以把 111 和 aka_kak​ 交换，使得两边独立的同时 2∣(fk−1+gk+1)2\mid(f_{k-1}+g_{k+1})2∣(fk−1​+gk+1​)，后手必败。

若 2∣(fk+gk+1)2\mid(f_k+g_{k+1})2∣(fk​+gk+1​)，则两个人都不能改变 fk+gk+1f_k+g_{k+1}fk​+gk+1​ 的奇偶性，不然另一个人交换 111 和 kkk 这个人就寄了。因此一定是换 aka_kak​ 和某个在 aka_kak​ 右边且奇偶性与 aka_kak​ 不同的数。

记 m=min⁡i>k{ai}m=\min_{i>k}\{a_i\}m=mini>k​{ai​}，那么除非 m>akm>a_km>ak​，此时先手换不了一点，后手获胜；否则最终肯定会把 aka_kak​ 换成 mmm，所以先手赢当且仅当换了奇数次，aka_kak​ 的奇偶性最终发生了变化，即一开始时 aka_kak​ 与 mmm 的奇偶性不同。

O(nlog⁡n)O(n\log n)O(nlogn)，瓶颈在于逆序对。

code 

#include <bits/stdc++.h>

using i64 = long long;

int main() {
	std::cin.tie(nullptr)->sync_with_stdio(false);
	int n, k;
	std::cin >> n >> k;
	k--;
	std::vector<int> a(n);
	for (int &i : a) std::cin >> i;
	int one = std::find(a.begin(), a.end(), 1) - a.begin();
	const auto inversion = [&](int l, int r, bool left) {
		std::vector<i64> fenwick(n + 1, 0);
		i64 res = 0;
		if (left) {
			for (int i = l; i <= r; i++) {
				for (int pos = a[i]; pos; pos ^= pos & -pos) res += fenwick[pos];
				for (int pos = a[i]; pos <= n; pos += pos & -pos) fenwick[pos]++;
			};
		} else {
			for (int i = r; i >= l; i--) {
				for (int pos = a[i]; pos; pos ^= pos & -pos) res += fenwick[pos];
				for (int pos = a[i]; pos <= n; pos += pos & -pos) fenwick[pos]++;
			};
		}
		return res;
	};
	if (one == k) {
		i64 total = inversion(0, k - 1, 1) + inversion(k + 1, n - 1, 0);
		std::cout << (total & 1 ? "Minchan\n" : "Junee\n");
		return 0;
	} else if (one < k) {
		i64 total = inversion(0, k, 1) + inversion(k + 1, n - 1, 0);
		if (total & 1) {
			std::cout << "Minchan\n";
			return 0;
		}
		int mi = *min_element(a.begin() + k + 1, a.end());
		if (mi > a[k]) {
			std::cout << "Junee\n";
			return 0;
		}
		std::cout << ((mi ^ a[k]) & 1 ? "Minchan\n" : "Junee\n");
		return 0;
	} else {
		i64 total = inversion(0, k - 1, 1) + inversion(k, n - 1, 0);
		if (total & 1) {
			std::cout << "Minchan\n";
			return 0;
		}
		int mi = *min_element(a.begin(), a.begin() + k + 1);
		if (mi > a[k]) {
			std::cout << "Junee\n";
			return 0;
		}
		std::cout << ((mi ^ a[k]) & 1 ? "Minchan\n" : "Junee\n");
		return 0;
	}
}

G

statement

给定 nnn 个物品，每个物品有三个价值 wi,vi,piw_i,v_i,p_iwi​,vi​,pi​，你要选其中 kkk 个，使得 ∑w+max⁡v+min⁡p\sum w+\max v+\min p∑w+maxv+minp 最小，并构造方案。

1≤k≤n≤2×1051\le k\le n\le2\times10^51≤k≤n≤2×105，0≤wi,vi,pi≤1090\le w_i,v_i,p_i\le10^90≤wi​,vi​,pi​≤109。

solution

按照 vvv 从小到大排序，枚举 vvv，那么问题变成：每次加一个物品 w,pw,pw,p，要求钦定这个物品必选的同时 ∑w+min⁡p\sum w+\min p∑w+minp 最小。

维护一个大小为 kkk 的集合 SSS，里面存的是当前的最优选择。初始时里面装满 (∞,∞)(\infin,\infin)(∞,∞)。

考虑加入一个物品 (w,p)(w,p)(w,p)，由于钦定其必选，很自然的想到贪心，把 SSS 中 www 最大的物品扔到换成这个。但是这样也有可能存在删去这个后 min⁡p\min pminp 反而变大使得答案不优的情况，此时删去 SSS 中 www 第二大的物品换掉即可，这样 min⁡p\min pminp 不会变大。当然也可以直接枚举这两种选择。

如果插入一个物品后答案比原来更劣，则可以在更新答案后直接把操作撤销回去。

输出方案是容易的。

O(nlog⁡n)O(n\log n)O(nlogn)，实现细节很多。

code 

#include <bits/stdc++.h>
 
using i64 = long long;
 
int main() {
	std::cin.tie(nullptr)->sync_with_stdio(false);
	int n, k;
	std::cin >> n >> k;
	std::vector<std::tuple<int, int, int, int>> a(n);
	for (int i = 0; i < n; i++) {
		auto &[w, v, p, id] = a[i];
		std::cin >> w >> v >> p;
		id = i;
	}
	if (k == 1) {
		std::pair<i64, int> ans = std::make_pair((i64)1e18, -1);
		for (auto [w, v, p, id] : a) {
			ans = std::min(ans, std::make_pair(0ll + w + v + p, id));
		}
		std::cout << ans.first << '\n' << ans.second + 1 << '\n';
		return 0;
	}
	auto b = a;
	std::sort(a.begin(), a.end(), [](auto x, auto y) {
		return std::get<1>(x) < std::get<1>(y);
	});
	std::set<std::pair<int, int>, std::greater<std::pair<int, int>>> ws;
	std::set<std::pair<int, int>> ps;
	for (int i = 0; i < k; i++) {
		b.emplace_back((int)2e9, (int)2e9, (int)2e9, n + i);
		ws.emplace((int)2e9, n + i);
		ps.emplace((int)2e9, n + i);
	}
	i64 cur = (i64)2e9 * (k + 1);
	auto ans = std::make_tuple((i64)1e18, -1, -1);
	for (auto [w, v, p, id] : a) {
		const auto del = [&](int idx) {
			i64 sum = cur - std::get<0>(b[idx]) - ps.begin()->first;
			ws.erase(std::make_pair(std::get<0>(b[idx]), idx));
			ps.erase(std::make_pair(std::get<2>(b[idx]), idx));
			ws.emplace(w, id);
			ps.emplace(p, id);
			sum += w + ps.begin()->first;
			int ret = ps.begin()->second;
			ws.erase(std::make_pair(w, id));
			ps.erase(std::make_pair(p, id));
			ws.emplace(std::get<0>(b[idx]), idx);
			ps.emplace(std::get<2>(b[idx]), idx);
			return std::make_tuple(sum, idx, ret);
		};
		int del1 = ws.begin()->second, del2 = next(ws.begin())->second;
		auto nw = std::min(del(del1), del(del2));
		ans = std::min(ans, std::make_tuple(get<0>(nw) + v, id, get<2>(nw)));
		if (get<0>(nw) < cur) {
			cur = get<0>(nw);
			ws.erase(std::make_pair(std::get<0>(b[get<1>(nw)]), get<1>(nw)));
			ps.erase(std::make_pair(std::get<2>(b[get<1>(nw)]), get<1>(nw)));
			ws.emplace(w, id);
			ps.emplace(p, id);
		}
	}
	auto [sum, mxv, mnp] = ans;
	std::cout << sum << '\n';
	std::vector<std::pair<int, int>> mx;
	for (auto [w, v, p, id] : a) {
		if (id != mxv && id != mnp) {
			if (v <= std::get<1>(b[mxv]) && p >= std::get<2>(b[mnp])) {
				mx.emplace_back(w, id);
			}
		}
	}
	std::sort(mx.begin(), mx.end());
	std::vector<int> res;
	for (int i = 0; i < (mxv == mnp ? k - 1 : k - 2); i++) {
		res.push_back(mx[i].second);
	}
	res.push_back(mxv);
	if (mxv != mnp) res.push_back(mnp);
	for (int i : res) std::cout << i + 1 << " \n"[i == res.back()];
}

H

statement

给定一个 nnn 个点的树，根为 111。有 mmm 个关键点，经过时你的钱会 −1-1−1。你要选择若干条边，选第 iii 条边的代价为 cic_ici​，使得经过这条边的时候你的钱 +1+1+1。对于任意的 iii，你要让你走到 1∼i1\sim i1∼i 时钱不为负数（一开始没钱），求你选择的边的代价和最小是多少，并构造方案。

1≤m<n≤2×1051\le m<n\le2\times10^51≤m<n≤2×105，0≤ci≤1090\le c_i\le10^90≤ci​≤109。

solution

考虑一开始只选关键点连向祖先的那条边，对于一条没选的边，如果选它可以不选下面的若干条不存在祖先关系的边。

对于一颗子树，维护若干个集合，其中第 iii 个表示除去 i−1i-1i−1 个集合所包含的边，选择的边的边权和最大的边集。那么选一条边一定是贪心的把第 111 个给换出来，用 std::multimap 维护即可。

两个子树合并时，相当于把两个 std::multimap 中对应位置合并，可以启发式合并。

O(nlog⁡2n)O(n\log^2n)O(nlog2n)，启发式合并细节很多，注意复杂度别写假了。

code 

#include <bits/stdc++.h>

using i64 = long long;

int main() {
	std::cin.tie(nullptr)->sync_with_stdio(false);
	int n, m;
	std::cin >> n >> m;
	std::vector<bool> key(n);
	for (int i = 0; i < m; i++) {
		int x;
		std::cin >> x;
		x--;
		key[x] = true;
	}
	std::vector<std::vector<std::tuple<int, int, int>>> g(n);
	for (int i = 0; i < n - 1; i++) {
		int u, v, w;
		std::cin >> u >> v >> w;
		u--, v--;
		g[u].emplace_back(v, w, i), g[v].emplace_back(u, w, i);
	}
	std::vector<std::multimap<i64, std::vector<int>, std::greater<i64>>> st(n);
	i64 sum = 0;
	std::set<int> res;
	const auto dfs = [&](auto &&self, int u, int f, int fw, int fi) -> void {
		for (auto [v, w, id] : g[u]) {
			if (v == f) continue;
			self(self, v, u, w, id);
			std::vector<std::pair<i64, std::vector<int>>> to_add;
			while (st[u].size() && st[v].size()) {
				if (st[u].begin()->second.size() < st[v].begin()->second.size()) {
					swap(st[u].begin()->second, st[v].begin()->second);
				}
				auto idx = std::move(st[u].begin()->second);
				for (int i : st[v].begin()->second) idx.push_back(i);
				to_add.emplace_back(st[u].begin()->first + st[v].begin()->first, std::move(idx));
				st[u].erase(st[u].begin()), st[v].erase(st[v].begin());
			}
			if (st[v].size()) st[u] = std::move(st[v]);
			for (auto &i : to_add) st[u].emplace(std::move(i));
		}
		if (key[u]) {
			sum += fw;
			st[u].emplace(fw, std::vector<int>({fi}));
			res.insert(fi);
		} else if (f != -1) {
			if (st[u].size() && fw < st[u].begin()->first) {
				sum += fw - st[u].begin()->first;
				for (int i : st[u].begin()->second) res.erase(i);
				res.insert(fi);
				st[u].erase(st[u].begin());
				st[u].emplace(fw, std::vector<int>({fi}));
			}
		}
	};
	dfs(dfs, 0, -1, 0, 0);
	std::cout << sum << '\n' << res.size() << '\n';
	for (int i : res) std::cout << i + 1 << " \n"[i == *--res.end()];
}

I

statement

你和对手在一棵边权为 111 的树上玩游戏，对手选定树上的某个节点不变，你每次可以询问一个点到对手选定的点的距离，问在最优策略下你最坏要几次才能唯一确定对手选定的节点。

2≤n≤2×1032\le n\le2\times10^32≤n≤2×103。

solution

假设第一次询问问了 rtrtrt 点，得到的距离为 ddd，那么相当于把这棵树以 rtrtrt 为根，对手的点深度为 ddd。

考虑 dp，设 fuf_ufu​ 表示当前已经确定在 uuu 的子树内时的最小步数，显然当 uuu 深度为 ddd 时有 fu=0f_u=0fu​=0，要求的即为 frtf_{rt}frt​。

容易发现，在 uuu 节点转移的时候，相当于要把包含儿子 v1,v2,⋯v_1,v_2,\cdotsv1​,v2​,⋯ 的 fff 值的序列 a=[fv1,fv2,⋯ ]a=[f_{v_1},f_{v_2},\cdots]a=[fv1​​,fv2​​,⋯] 进行重排，使得 ai+(i−1)a_i+(i-1)ai​+(i−1) 最小（这里 (i−1)(i-1)(i−1) 是前面儿子尝试错误的代价），可以证明将 aaa 降序排序最优。

考虑优化，由于一个位置的 dp 只和对手选的点在其子树中的深度有关，而即使在不定根的情况下，子树也只有 2n−22n-22n−2 个（考虑一颗子树实质上是断开一条边形成的，而边只有 n−1n-1n−1 条），因此只要按深度从小到大 dpdpdp 即可。

瓶颈在于排序，时间复杂度 O(n2log⁡n)O(n^2\log n)O(n2logn)。

code 

#include <bits/stdc++.h>

int main() {
	std::cin.tie(nullptr)->sync_with_stdio(false);
	int n;
	std::cin >> n;
	std::vector<std::vector<int>> g(n);
	for (int i = 0; i < n - 1; i++) {
		int u, v;
		std::cin >> u >> v;
		u--, v--;
		g[u].push_back(v), g[v].push_back(u);
	}
	std::vector<std::vector<std::vector<int>>> dp(n, std::vector<std::vector<int>>(n));
	for (int i = 0; i < n; i++) {
		for (int v : g[i]) {
			dp[i][v].assign(n, -1);
			dp[i][v][0] = 0;
		}
	}
	std::vector<int> ans(n), pos(n);
	for (int i = 1; i < n; i++) {
		for (int rt = 0; rt < n; rt++) {
			std::vector<int> cur;
			for (int v : g[rt]) {
				if (dp[v][rt][i - 1] != -1) {
					cur.push_back(dp[v][rt][i - 1]);
				}
			}
			std::sort(cur.begin(), cur.end(), std::greater());
			for (int j = 0; j < (int)cur.size(); j++) {
				pos[cur[j]] = j;
			}
			std::vector<int> pmx(cur.size()), smx(cur.size());
			for (int j = 0; j < (int)cur.size(); j++) {
				pmx[j] = cur[j] + j;
				if (j) {
					pmx[j] = std::max(pmx[j - 1], pmx[j]);
				}
			}
			for (int j = (int)cur.size() - 1; j >= 0; j--) {
				smx[j] = cur[j] + j;
				if (j < (int)cur.size() - 1) {
					smx[j] = std::max(smx[j + 1], smx[j]);
				}
			}
			int mx = pmx.size() ? pmx.back() : -1;
			ans[rt] = std::max(ans[rt], mx);
			for (int v : g[rt]) {
				if (dp[v][rt][i - 1] == -1) {
					dp[rt][v][i] = mx;
				} else {
					dp[rt][v][i] = -1;
					int p = pos[dp[v][rt][i - 1]];
					if (p) {
						dp[rt][v][i] = std::max(dp[rt][v][i], pmx[p - 1]);
					}
					if (p < (int)cur.size() - 1) {
						dp[rt][v][i] = std::max(dp[rt][v][i], smx[p + 1] - 1);
					}
				}
			}
		}
	}
	std::cout << *std::min_element(ans.begin(), ans.end()) + 1 << '\n';
}

statement

给定一个 nnn 个点 mmm 条边的无向简单联通图，第 iii 条边有边权 cic_ici​，对于每个 1≤i≤k1\le i\le k1≤i≤k 的 iii，回答从 1∼n1\sim n1∼n 恰好经过 iii 个点（可经过重复点或边，但不能在一条边上中途折返，含起点和终点）的最短路，若无法到达则答案为 000，求答案的和  mod 998244353\bmod998244353mod998244353 的值。

2≤n≤1.5×103,n−1≤m≤3×103,1≤k,ci≤1092\le n\le1.5\times10^3,n-1\le m\le3\times10^3,1\le k,c_i\le10^92≤n≤1.5×103,n−1≤m≤3×103,1≤k,ci​≤109。

solution

令 k′=k−1k'=k-1k′=k−1，则原题相当于经过恰好 k′k'k′ 条边的最短路。

考虑最优方案一定是：从 111 走到某个点，然后在跟这个点相连的边权最小的边上反复横跳，最后再从这个点走到 nnn。由于反复横跳一次要经过两条边，启示我们要分奇偶讨论。

枚举该点 uuu，设跟这个点相连的边权最小的边的边权为 www，然后将整个图所有边边权 −w-w−w，在不走负权边的情况下，设 1∼u1\sim u1∼u 和 u∼nu\sim nu∼n 的奇、偶最短路分别为 do,1,de,1,do,n,de,nd_{o,1},d_{e,1},d_{o,n},d_{e,n}do,1​,de,1​,do,n​,de,n​，则最终 k′k'k′ 为奇数的答案为 wk′+min⁡(do,1+de,n,de,1+do,n)wk'+\min(d_{o,1}+d_{e,n},d_{e,1}+d_{o,n})wk′+min(do,1​+de,n​,de,1​+do,n​)，k′k'k′ 为偶数的答案为 wk′+min⁡(do,1+do,n,de,1+de,n)wk'+\min(d_{o,1}+d_{o,n},d_{e,1}+d_{e,n})wk′+min(do,1​+do,n​,de,1​+de,n​)。

观察到答案是关于 k′k'k′ 的若干条直线的最下面的部分，直接把这若干条线段拉出来求和即可。

然而，当 k′k'k′ 较小时，我们有可能走 <0<0<0 次反复横跳的边，因此 k′k'k′ 较小时需要直接暴力 dp 求解，k′k'k′ 较大时再用上面的方法。一个较紧的分界点是 2n2n2n。

总时间复杂度 O(nmlog⁡(n+m))O(nm\log(n+m))O(nmlog(n+m))，理论上可以把 log⁡\loglog 优化掉，但没必要。

code 

#include <bits/stdc++.h>

using i64 = long long;

constexpr int mod = 998244353;

int main() {
	std::cin.tie(nullptr)->sync_with_stdio(false);
	int n, m, k;
	std::cin >> n >> m >> k;
	k--;
	std::vector<std::vector<std::pair<int, int>>> g(n);
	for (int i = 0; i < m; i++) {
		int u, v, w;
		std::cin >> u >> v >> w;
		u--, v--;
		g[u].emplace_back(v, w), g[v].emplace_back(u, w);
	}
	int ans = 0;
	int steps = std::min(k, n + n);
	std::vector<std::vector<i64>> dp(n, std::vector<i64>(steps + 1, 1e18));
	dp[0][0] = 0;
	for (int i = 1; i <= steps; i++) {
		for (int u = 0; u < n; u++) {
			for (auto [v, w] : g[u]) dp[u][i] = std::min(dp[u][i], dp[v][i - 1] + w);
		}
		if (dp[n - 1][i] != 1e18) ans = (ans + dp[n - 1][i]) % mod;
	}
	if (steps == k) {
		std::cout << ans << '\n';
		return 0;
	}
	std::array<std::vector<std::pair<int, i64>>, 2> lines;
	for (int u = 0; u < n; u++) {
		int weight = 2e9;
		for (auto [v, w] : g[u]) weight = std::min(weight, w);
		std::array<std::vector<i64>, 2> dis;
		std::array<std::vector<bool>, 2> vis;
		for (auto &i : dis) i.assign(n, 1e18);
		for (auto &i : vis) i.assign(n, false);
		std::priority_queue<std::tuple<i64, int, int>> q;
		dis[0][u] = 0;
		q.emplace(0, 0, u);
		while (q.size()) {
			auto [dist, parity, cur] = q.top();
			q.pop();
			if (!vis[parity][cur]) {
				vis[parity][cur] = true;
				for (auto [v, w] : g[cur]) {
					int wei = w - weight;
					if (wei >= 0 && dis[!parity][v] > dis[parity][cur] + wei) {
						dis[!parity][v] = dis[parity][cur] + wei;
						q.emplace(-dis[!parity][v], !parity, v);
					}
				}
			}
		}
		for (int p = 0; p < 2; p++) {
			i64 mn = 1e18;
			for (int q = 0; q < 2; q++) {
				mn = std::min(mn, dis[q][0] + dis[p ^ q][n - 1]);
			}
			if (mn != 1e18) lines[p].emplace_back(weight, mn);
		}
	}
	for (int p = 0; p < 2; p++) {
		const auto calc = [&](std::pair<int, i64> line, int x) {
			return 1ll * line.first * x + line.second;
		};
		const auto intersection = [&](std::pair<int, i64> line1, std::pair<int, i64> line2) -> int {
			if (line1.first == line2.first) return 2e9;
			return std::min((i64)2e9, (line2.second - line1.second) / (line1.first - line2.first));
		};
		int bg = steps + 1;
		if ((bg & 1) != p) bg++;
		while (true) {
			std::pair<int, i64> cur;
			i64 y = 1e18;
			for (auto i : lines[p]) {
				i64 val = calc(i, bg);
				if (y > val || (y == val && i.first < cur.first)) {
					y = val;
					cur = i;
				}
			}
			if (y == 1e18) break;
			int ed = k;
			for (auto i : lines[p]) {
				int is = intersection(cur, i);
				if (is >= bg) ed = std::min(ed, is);
			}
			if ((ed & 1) != p) ed--;
			if (bg <= ed) {
				int cnt = (ed - bg) / 2 + 1;
				ans = (ans + (mod + 1ll) / 2 * cur.first % mod * (bg + ed) % mod * cnt) % mod;
				ans = (ans + cur.second % mod * cnt) % mod;
			}
			bg = ed + 2;
			if (bg > k) break;
		}
	}
	std::cout << ans << '\n';
}
